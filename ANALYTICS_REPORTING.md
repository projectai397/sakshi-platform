### Database Queries

**Example: Monthly Revenue by Tier**
```sql
SELECT 
  DATE_FORMAT(created_at, '%Y-%m') as month,
  price_tier,
  COUNT(*) as orders,
  SUM(total) as revenue
FROM sakshi_cafe_orders
WHERE created_at >= DATE_SUB(NOW(), INTERVAL 12 MONTH)
GROUP BY month, price_tier
ORDER BY month DESC, price_tier;
```

**Example: Top Menu Items**
```sql
SELECT 
  mi.title,
  COUNT(oi.id) as times_ordered,
  SUM(oi.quantity) as total_quantity,
  AVG(oi.price) as avg_price
FROM sakshi_menu_items mi
JOIN order_items oi ON mi.id = oi.menu_item_id
WHERE oi.created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)
GROUP BY mi.id
ORDER BY times_ordered DESC
LIMIT 10;
```

**Example: User Retention**
```sql
WITH cohorts AS (
  SELECT 
    user_id,
    DATE_FORMAT(MIN(created_at), '%Y-%m') as cohort_month
  FROM sakshi_cafe_orders
  GROUP BY user_id
)
SELECT 
  cohort_month,
  COUNT(DISTINCT c.user_id) as cohort_size,
  COUNT(DISTINCT CASE 
    WHEN o.created_at >= DATE_ADD(c.cohort_month, INTERVAL 1 MONTH)
    THEN o.user_id 
  END) as retained_users,
  ROUND(COUNT(DISTINCT CASE 
    WHEN o.created_at >= DATE_ADD(c.cohort_month, INTERVAL 1 MONTH)
    THEN o.user_id 
  END) * 100.0 / COUNT(DISTINCT c.user_id), 2) as retention_rate
FROM cohorts c
LEFT JOIN sakshi_cafe_orders o ON c.user_id = o.user_id
GROUP BY cohort_month
ORDER BY cohort_month DESC;
```
